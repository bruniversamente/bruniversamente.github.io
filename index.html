Pergunta: onde a margem cai antes da receita parecer um problema?

import pandas as pd
from pathlib import Path
base = Path("data")
arquivo = base / "mart_vendas.parquet"
vendas = pd.read_parquet(arquivo)
lucro = vendas["lucro"]
receita = vendas["receita"]
meta = vendas["meta_margem"]
vendas["margem"] = lucro / receita
vendas["gap"] = vendas["margem"] - meta
risco = vendas[vendas.gap < 0]
resumo = risco.groupby("canal").agg(
    receita=("receita", "sum"),
    margem=("margem", "mean"),
)
saida = "outputs/canais_em_risco.csv"
resumo.to_csv(saida)
